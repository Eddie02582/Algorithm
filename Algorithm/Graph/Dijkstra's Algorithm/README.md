import heapq

def dijkstra(n, edges, start):
    """
    n: ç¯€é»æ•¸é‡ (0 ~ n-1)
    edges: é„°æ¥è¡¨ {u: [(v, w), ...], ...}

    dist[start] = 0
    heap = [(0, start)]
    
    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]:
            continue
        for v, w in edges.get(u, []):
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(heap, (dist[v], v))
    return dist
